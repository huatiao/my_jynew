using Loxodon.Framework.Interactivity;
using ProjectBase.Define;

namespace #NAMESPACE#
{
    [UIInfo(UIID.#UIID#)]
    public class #CLASSNAME# : VMWindow<#VIEWMODEL#>
    {
        protected override void OnViewCreate()
        {
            BindingSet<#CLASSNAME#, #VIEWMODEL#> bindingSet = this.CreateBindingSet<#CLASSNAME#, #VIEWMODEL#>();
            
            bindingSet.Bind().For(v => v.OnInteractDismissed).To(vm => vm.InteractDismissed);

        {
            Dismiss();
        }
    }
